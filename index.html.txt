    <script>
        const canvas = document.getElementById('cosmicCanvas');
        const ctx = canvas.getContext('2d');

        let width = canvas.width = window.innerWidth;
        let height = canvas.height = window.innerHeight;

        window.addEventListener('resize', () => {
            width = canvas.width = window.innerWidth;
            height = canvas.height = window.innerHeight;
        });

        const particleCount = 850; 
        const heartParticleCount = 550; 
        const particles = [];
        const heartParticles = [];
        
        // Coordenadas lógicas del centro del universo
        let targetCenterX = width / 2;
        let targetCenterY = height / 2;
        let currentCenterX = width / 2;
        let currentCenterY = height / 2;
        
        let universoIniciado = false;
        let tiempoTranscurrido = 0;
        let fraseActualIndex = 0; 
        
        // Variables para controlar el arrastre táctil / mouse
        let estaPresionando = false;
        let inicioX = 0;
        let inicioY = 0;

        class GalacticParticle {
            constructor() {
                this.reset();
            }

            reset() {
                this.angle = Math.random() * Math.PI * 2;
                this.radius = 0; 
                
                this.explosionSpeed = Math.random() * 18 + 6; 
                this.speed = Math.random() * 1.8 + 0.8;
                this.radialSpeed = Math.random() * 0.015 + 0.004;
                this.size = Math.random() * 2.2 + 0.8;
                
                const colors = ['#00bfff', '#7f00ff', '#ff00aa', '#00ffcc', '#ffcc00', '#ffffff', '#ff0055'];
                this.color = colors[Math.floor(Math.random() * colors.length)];
                this.fase = 'explosion';
            }

            update() {
                if (this.fase === 'explosion') {
                    this.radius += this.explosionSpeed;
                    this.explosionSpeed *= 0.94;

                    if (this.explosionSpeed < 0.15) {
                        this.fase = 'espiral';
                    }
                } else {
                    this.radius -= this.speed * 0.6;
                    this.angle += this.radialSpeed;

                    if (this.radius <= 6) {
                        this.radius = Math.random() * (Math.max(width, height) / 2) + 150;
                        this.fase = 'espiral';
                    }
                }
            }

            draw() {
                const x = currentCenterX + Math.cos(this.angle) * this.radius;
                const y = currentCenterY + Math.sin(this.angle) * this.radius * 0.48; 

                ctx.beginPath();
                ctx.arc(x, y, this.size, 0, Math.PI * 2);
                ctx.fillStyle = this.color;
                
                ctx.shadowBlur = 10;
                ctx.shadowColor = this.color;
                
                ctx.fill();
            }
        }

        function getHeartPoint(t) {
            const x = 16 * Math.pow(Math.sin(t), 3);
            const y = 13 * Math.cos(t) - 5 * Math.cos(2*t) - 2 * Math.cos(3*t) - Math.cos(4*t);
            return { x: x, y: -y };
        }

        class HeartParticle {
            constructor() {
                this.t = Math.random() * Math.PI * 2;
                const point = getHeartPoint(this.t);
                
                this.scale = Math.random() * 3 + 7.5; 
                this.offsetX = point.x * this.scale;
                this.offsetY = point.y * this.scale;
                
                this.x = width / 2;
                this.y = height / 2;
                
                this.vx = (Math.random() - 0.5) * (Math.random() * 35 + 15);
                this.vy = (Math.random() - 0.5) * (Math.random() * 35 + 15);
                
                this.size = Math.random() * 2.2 + 0.9;
                
                const heartColors = ['#ffff00', '#ffd700', '#ffffff', '#ffea00', '#f4d03f'];
                this.color = heartColors[Math.floor(Math.random() * heartColors.length)];
                
                this.speed = Math.random() * 0.025 + 0.008;
                this.fase = 'explosion';
            }

            update() {
                const targetX = currentCenterX + this.offsetX;
                const targetY = currentCenterY + this.offsetY;

                if (this.fase === 'explosion') {
                    this.x += this.vx;
                    this.y += this.vy;
                    this.vx *= 0.92;
                    this.vy *= 0.92;

                    if (Math.abs(this.vx) < 0.1 && Math.abs(this.vy) < 0.1) {
                        this.fase = 'formarCorazon';
                    }
                } else {
                    this.x += (targetX - this.x) * this.speed;
                    this.y += (targetY - this.y) * this.speed;
                }
            }

            draw() {
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
                ctx.fillStyle = this.color;
                
                ctx.shadowBlur = 22;
                ctx.shadowColor = '#ffd700';
                
                ctx.fill();
            }
        }

        for (let i = 0; i < particleCount; i++) particles.push(new GalacticParticle());
        for (let i = 0; i < heartParticleCount; i++) heartParticles.push(new HeartParticle());

        function animate() {
            ctx.globalCompositeOperation = 'destination-out';
            ctx.fillStyle = 'rgba(0, 0, 0, 0.15)';
            ctx.fillRect(0, 0, width, height);
            
            ctx.globalCompositeOperation = 'source-over';

            // Suavizado del movimiento del centro (Efecto inercia/amortiguación)
            if (!estaPresionando) {
                targetCenterX = width / 2;
                targetCenterY = height / 2;
            }
            currentCenterX += (targetCenterX - currentCenterX) * 0.08;
            currentCenterY += (targetCenterY - currentCenterY) * 0.08;

            if (universoIniciado) {
                tiempoTranscurrido++;

                if (tiempoTranscurrido > 40) {
                    const brillo = Math.min((tiempoTranscurrido - 40) / 120, 1);
                    ctx.shadowBlur = 0; 
                    
                    let gradient = ctx.createRadialGradient(currentCenterX, currentCenterY, 0, currentCenterX, currentCenterY, 65);
                    gradient.addColorStop(0, '#ffffff');
                    gradient.addColorStop(0.2, 'rgba(255, 215, 0, 0.35)');
                    gradient.addColorStop(0.6, `rgba(255, 234, 0, ${brillo * 0.2})`);
                    gradient.addColorStop(1, 'transparent');
                    ctx.fillStyle = gradient;
                    ctx.beginPath();
                    ctx.arc(currentCenterX, currentCenterY, 65, 0, Math.PI * 2);
                    ctx.fill();
                }

                particles.forEach(p => {
                    p.update();
                    p.draw();
                });

                heartParticles.forEach(hp => {
                    hp.update();
                    hp.draw();
                });

                if (tiempoTranscurrido === 130) {
                    ctx.shadowBlur = 0;
                    document.getElementById('textoFinal').style.opacity = '1';
                }
                if (tiempoTranscurrido === 190) {
                    document.getElementById('frase1').classList.add('activa');
                    fraseActualIndex = 1; 
                    document.getElementById('pistaClick').style.opacity = '1';
                }
            }

            requestAnimationFrame(animate);
        }

        // --- SISTEMA INTERACTIVO DE ARRASTRE (MOUSE Y TÁCTIL) ---
        function presionarInicio(x, y) {
            if (!universoIniciado) return;
            estaPresionando = true;
            inicioX = x;
            inicioY = y;
        }

        function moverMover(x, y) {
            if (!estaPresionando) return;
            const deltaX = x - inicioX;
            const deltaY = y - inicioY;
            
            // Movemos el centro relativo a la pantalla
            targetCenterX = (width / 2) + deltaX;
            targetCenterY = (height / 2) + deltaY;
        }

        function soltarFin() {
            estaPresionando = false;
        }

        // Eventos para Mouse
        window.addEventListener('mousedown', (e) => {
            if (e.target.id === 'btnPlay') return;
            presionarInicio(e.clientX, e.clientY);
        });
        window.addEventListener('mousemove', (e) => {
            moverMover(e.clientX, e.clientY);
        });
        window.addEventListener('mouseup', () => {
            soltarFin();
        });

        // Eventos para Pantallas Táctiles (Celulares / Tablets)
        window.addEventListener('touchstart', (e) => {
            if (e.target.id === 'btnPlay') return;
            const toque = e.touches[0];
            presionarInicio(toque.clientX, toque.clientY);
        });
        window.addEventListener('touchmove', (e) => {
            const toque = e.touches[0];
            moverMover(toque.clientX, toque.clientY);
        });
        window.addEventListener('touchend', () => {
            soltarFin();
        });


        // --- SISTEMA DE NAVEGACIÓN DE FRASES CON CLICS COMPLETOS ---
        window.addEventListener('click', (e) => {
            if (e.target.id === 'btnPlay') return;
            
            // Solo cambia de frase si el usuario hace un clic corto (sin arrastrar demasiado)
            if (fraseActualIndex >= 1 && fraseActualIndex < 5 && !estaPresionando) {
                const fraseActualElemento = document.getElementById(`frase${fraseActualIndex}`);
                const siguienteFraseElemento = document.getElementById(`frase${fraseActualIndex + 1}`);
                
                if (fraseActualElemento && siguienteFraseElemento) {
                    fraseActualElemento.classList.remove('activa');
                    fraseActualElemento.classList.add('oculta');
                    
                    if (fraseActualIndex === 4) { 
                        document.getElementById('pistaClick').style.opacity = '0';
                    }
                    
                    setTimeout(() => {
                        siguienteFraseElemento.classList.add('activa');
                        fraseActualIndex++;
                    }, 500);
                }
            }
        });

        document.getElementById('btnPlay').addEventListener('click', (e) => {
            e.stopPropagation(); 
            document.getElementById('pantallaInicio').classList.add('hidden');
            universoIniciado = true;
            
            const audio = document.getElementById('musicaFondo');
            audio.play().catch(error => console.log("Audio bloqueado:", error));
        });

        animate();
    </script>
